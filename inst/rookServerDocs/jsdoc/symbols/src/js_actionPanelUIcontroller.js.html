  6 /**
  7  * Manages the action panel interface. Singleton
  8  * @constructor
  9  */
 10 function actionPanelUIcontroller() {
 11     if (typeof actionPanelUIcontroller.instance === 'object') {
 12 	return actionPanelUIcontroller.instance;
 13     }
 14     
 15     this.generateCellSelectionStore();
 16     this.generateUI();
 17 
 18     // Setup listener for selection change
 19     var evtBus = new eventBus();
 20     evtBus.register("cell-selection-updated", null, function() {
 21 	var actPaneUICntr = new actionPanelUIcontroller();
 22 	actPaneUICntr.syncCellSelectionStore();
 23     });
 24 
 25     actionPanelUIcontroller.instance = this; 
 26 };

 29 /**
 30  * Generate the UI controls
 31  * @private
 32  */
 33 actionPanelUIcontroller.prototype.generateUI = function() {
 34     var actionsTab = Ext.getCmp('actions-ui-tab');
 35     
 36     var actionsInnerTab = Ext.create('Ext.TabPanel', {
 37 	layout: 'fit',
 38 	width: '100%',
 39 	height: '100%',
 40 	tabBarPosition: 'top',
 41 	items: [
 42 	    {
 43 		layout: 'fit',
 44 		title: 'Differential Expression',
 45 		id: 'differentialExpressionTab',
 46 		glyph: 0xf14e, //fa-compass
 47 		tooltip: 'Perform differential expression between two groups of cells',
 48 		height: '100%',
 49 		width: '100%',
 50 	    },
 51 	    {
 52 		layout: 'fit',
 53 		title: 'Enrichment',
 54 		glyph: 0xf200, // pie chart
 55 		tooltip: 'Perform enrichment analysis',
 56 		height: '100%',
 57 		width: '100%'
 58 	    }
 59 	]
 60     });
 61 
 62     // Data store for the available diff expr methods
 63     var deMethodsStore = Ext.create('Ext.data.Store', {
 64 	id: 'availableDEmethodStore',
 65 	fields: [{name: 'name', type: 'string'},
 66 	       {name:'displayname', type: 'string'}]
 67     });
 68 
 69     // Populate the store
 70     var calcFactory = new calculationControllerFactory();
 71     var calcCntr = calcFactory.makeCalculationController();
 72     var availableMethods = calcCntr.getAvailableDEmethods();
 73     for (i in availableMethods) {
 74 
 75 	deMethodsStore.add({name: availableMethods[i].name, displayname: availableMethods[i].displayName});
 76     }
 77 
 78     var deTab = Ext.getCmp('differentialExpressionTab');
 79 
 80     var formPanelDE =  Ext.create('Ext.form.Panel', {
 81 	id: 'formPanelDE',
 82 	height: '100%',
 83 	width: '100%',
 84 	bodyPadding: 10,
 85 	items: [{
 86 	    id: 'selectionA',
 87 	    xtype: 'combo',
 88 	    fieldLabel: 'Cell Selection A',
 89 	    queryMode: 'local',
 90 	    editable: false,
 91 	    store: Ext.data.StoreManager.lookup('cellSelectionStoreForDE'),
 92 	    displayField: 'displayname',
 93 	    valueField: 'selectionname'
 94 	},
 95 	{
 96 	    id: 'selectionB',
 97 	    xtype: 'combo',
 98 	    fieldLabel: 'Cell Selection B',
 99 	    queryMode: 'local',
100 	    editable: false,
101 	    store: Ext.data.StoreManager.lookup('cellSelectionStoreForDE'),
102 	    displayField: 'displayname',
103 	    valueField: 'selectionname'
104 	},{
105 	    id: 'selectionMethod',
106 	    xtype: 'combo',
107 	    fieldLabel: 'Method', // TODO add help pane
108 	    queryMode: 'local',
109 	    editable: 'false',
110 	    store: Ext.data.StoreManager.lookup('availableDEmethodStore'),
111 	    displayField: 'displayname',
112 	    valueField: 'name'
113 	},{
114 	    xtype: 'button',
115 	    text: 'Run differential expression',
116 	    handler: function() {	
117 		var selectionA = Ext.getCmp("formPanelDE").getForm()
118 		    .findField("selectionA").getValue();
119 		var selectionB = Ext.getCmp("formPanelDE").getForm()
120 		    .findField("selectionB").getValue();
121 
122 		if (selectionA === selectionB) {
123 		    Ext.MessageBox.alert('Warning', 'Please select a differnt set for A and B');
124 		} else {
125 		    //TODO: Some kind of visual wait indicator
126 		    var calcFactory = new calculationControllerFactory();
127 		    var calcCntr = calcFactory.makeCalculationController();
128 
129 		    calcCntr.performDEbySelection(selectionA, selectionB, function() {} );
130 		}
131 	    }
132 	}
133 	] //items
134     });
135     deTab.add(formPanelDE);
136 
137 
138     actionsTab.add(actionsInnerTab);
139 };

141 /**
142  * Generate the cell selection store for populating the dropdowns
143  * @private
144  */
145 actionPanelUIcontroller.prototype.generateCellSelectionStore = function() {
146     Ext.create('Ext.data.Store', {
147 	storeId: 'cellSelectionStoreForDE',
148 	fields: [
149 		{ name: 'selectionname', type: 'string'},
150 		{ name: 'displayname', type: 'string'}
151 	],
152 	autoLoad: true
153     });
154 }

157 /**
158  * Update the cell selection store for populating the dropdowns
159  * with information from the selection controller
160  * @private
161  */
162 actionPanelUIcontroller.prototype.syncCellSelectionStore = function() {
163     // Get the store for the table and empty it
164     var store = Ext.data.StoreManager.lookup('cellSelectionStoreForDE');
165     store.removeAll();
166 
167     // Repopulate the store
168     var cellSelCntr =  new cellSelectionController();
169     var availSelections = cellSelCntr.getAvailableSelections();
170     
171     for (sel in availSelections) {
172 	var selName = availSelections[sel];;
173 	var selDisplayName =  cellSelCntr.getSelectionDisplayName(selName);
174 
175 	store.add({selectionname: selName, displayname: selDisplayName});
176     }// for
177 }
