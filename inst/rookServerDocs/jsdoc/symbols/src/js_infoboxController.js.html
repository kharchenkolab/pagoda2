  2 /**
  3  * Manages the information box and its content
  4  * @constructor
  5  */
  6 function infoboxController() {
  7     if (typeof infoboxController.instance === 'object') {
  8 	return infoboxController.instance;
  9     }   
 10 
 11     var evtBus = new eventBus();
 12 
 13     console.log("Initializing infobox controller...");
 14 
 15     // Generate the table and a store for it
 16     var infoAppContainer = Ext.getCmp('information-app-container');
 17     var messageTableStore = Ext.create('Ext.data.Store', {
 18 	id: 'messageTableStore',
 19 	sortable: false,
 20 	defaultSortable: false,
 21 	fields: [{name: 'message', type: 'string'}],
 22     });
 23     
 24     var msgTable = Ext.create('Ext.grid.Panel', {
 25 	store: Ext.data.StoreManager.lookup('messageTableStore'),
 26 	columns: [{ text: 'Messages', dataIndex: 'message', width: '100%', sortable: false }],
 27 	width: '100%',
 28 	height: '100%',
 29 	scrollable: true
 30     });
 31 
 32     infoAppContainer.add(msgTable);
 33 
 34     // Register Events to update the Messages as required here
 35 
 36     // Report update of cell seletion
 37     evtBus.register("cell-selection-updated", null, function() {
 38 	var selCntr = new cellSelectionController();
 39 	var primarySel = selCntr.getSelection('currentPrimarySel');
 40 	var secondarySel = selCntr.getSelection('currentAltSel');
 41 
 42 	var infBoxCntr = new infoboxController();
 43 	infBoxCntr.addMessage('--- Selection Updated ---' );
 44 	if ( typeof primarySel != 'undefined' ) 
 45 	    infBoxCntr.addMessage('Primary cell selection: ' + primarySel.length);
 46 
 47 	if( typeof secondarySel !== 'undefined')
 48 	    infBoxCntr.addMessage('Secondary cell selection: ' + secondarySel.length);
 49     });
 50 
 51     // Singleton stuff
 52     infoboxController.instance =  this;
 53 }
 54 
 55 /**
 56  * Add a message to the panel
 57  */
 58 infoboxController.prototype.addMessage = function(msg) {
 59     var store = Ext.data.StoreManager.lookup('messageTableStore')
 60     store.add({message: msg});
 61 }
